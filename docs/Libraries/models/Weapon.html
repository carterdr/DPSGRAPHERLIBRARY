 96    def calculate(self, buff_perc=1.25, custom_name=None, prev_result=None, 
 97                  primary_damage=0, special_damage=0, p_to_s=default_swap_time, 
 98                  s_to_h=default_swap_time, h_to_p=default_swap_time, pre_bait_charge_time = 0, **kwargs):
 99        """
100        Runs the appropriate damage loop simulation based on the weapon's `damage_loop_type`.
101
102        Args:
103            buff_perc (float): Buff percentage to apply (default = 1.25).
104            custom_name (str): Optional name override for the result.
105            prev_result (DamageResult): Optional previous result for chaining.
106            primary_damage, special_damage (float): Used in bait loops to simulate cross-weapon procs.
107            p_to_s (float): Swap time from first weapon to second weapon
108            s_to_h (float): Swap time from second weapon to heavy weapon (the weapon that will be mag dumped) 
109            h_to_p (float): Swap time from heavy weapon to first weapon
110            pre_bait_charge_time (float): Optional pre-bait charge time (such as using a fusion or linear as the first weapon).
111            **kwargs: Additional context for custom damage functions.
112
113        Returns:
114            DamageResult: The final simulation output as a 1001-length time-damage array.
115        """
116        if custom_name is None:
117            name = self.name  # Default to class name if not provided
118        else:
119            name = custom_name
120        self._prepare_calculation(prev_result)
121        """ 
122        if self.charge_time != 0 and self.sim_state.time != 0:
123        self.sim_state.time -= 0.5
124        """
125        self.sim_state.time += self.charge_time
126        # incase we are doing bait and we have a charge_time primary like riptide
127        bait_tuple = [
128            (p_to_s, primary_damage * buff_perc),
129            (s_to_h, special_damage * buff_perc),
130            (h_to_p, 0)
131        ]
132        self.sim_state.time += pre_bait_charge_time
133
134        # Select appropriate damage loop
135        loop_type = self.damage_loop_type
136        if loop_type == "simple":
137            def wrapped_damage_function():
138                return self.damage_per_shot_function(buff_perc, **kwargs)
139            self.processSimpleDamageLoop(wrapped_damage_function)
140        elif loop_type == "bait":
141            def wrapped_damage_function(is_proc_shot):
142                return self.damage_per_shot_function_bait(is_proc_shot, buff_perc, **kwargs)
143            self.processBaitDamageLoop(bait_tuple, wrapped_damage_function, kwargs.get("dont_reproc", False))
144        elif loop_type == "envious":
145            def wrapped_damage_function(is_proc_shot):
146                return self.damage_per_shot_function_bait(is_proc_shot, buff_perc, **kwargs)
147            self.processEnviousArsenalBaitDamageLoop(bait_tuple, wrapped_damage_function)
148        elif loop_type == "refund":
149            def wrapped_damage_function():
150                return self.damage_per_shot_function(buff_perc, **kwargs)
151            self.processRefundLoop(wrapped_damage_function)
152        return self.fill_gaps(self.sim_state.damage_times, name, self.category)
153    def processSimpleDamageLoop(self, damage_per_shot_function: Callable, 
154                                special_reload_function : Callable = None, after_x_do: Callable = None, 
155                                x : int = 0, proc_progress : int = 1,
156                                after_y_do: Callable = None, y : int = 0, special_reload_function_y: Callable = None):
157            """
158            Processes a standard damage loop where the weapon reloads after emptying the magazine.
159
160            Args:
161                damage_per_shot_function (Callable): Function returning damage per shot.
162                special_reload_function (Callable): Optional function to call instead of default reload (add reload time). 
163                                                    Also runs if ammo runs out before a reload so it works with Grand overture.
164                after_x_do (Callable): Optional function triggered every X hits.
165                x (int): Number of hits before triggering `after_x_do`.
166                proc_progress (int): Increment toward `x` after each shot.
167                after_y_do (Callable): Optional second trigger function.
168                y (int): Hits before `after_y_do`.
169                special_reload_function_y (Callable): Secondary custom reload behavior that only runs when we reload, necessary for reseting y refund shots.
170            """
171            
172            self.sim_state.mag_size = self.mag_size_initial
173            self.sim_state.mag  = 0
174            hits_to_proc_x = x
175            hits_to_proc_y = y
176            while self.sim_state.shots_fired < self.reserves and self.sim_state.time < self.MAX_SIM_TIME:
177                self.sim_state.shots_fired_this_mag = 0
178                while self.sim_state.shots_fired_this_mag < self.sim_state.mag_size:
179                    dmg = damage_per_shot_function()
180                    self.sim_state.damage_done += dmg
181                    self.sim_state.shots_fired += 1
182                    self.sim_state.shots_fired_this_mag += 1
183                    hits_to_proc_x -= proc_progress
184                    hits_to_proc_y -= proc_progress
185                    self.sim_state.damage_times.append(self.update())
186                    if hits_to_proc_x <= 0:
187                        if after_x_do:
188                            if print_update:
189                                print(f"      - After_X_Proc {hits_to_proc_x}")
190                            after_x_do()
191                            hits_to_proc_x += x
192                    if hits_to_proc_y <= 0:
193                        if after_y_do:
194                            if print_update:
195                                print(f"      - After_Y_Proc {hits_to_proc_y}")
196                            after_y_do()
197                            hits_to_proc_y += y
198                    if self.sim_state.shots_fired == self.reserves:
199                        if special_reload_function:
200                            hits_to_proc_x = special_reload_function() or 0 # Refund function to reset after reload
201                        break
202                    # Reload logic
203                    if self.sim_state.shots_fired_this_mag == self.sim_state.mag_size:
204                        if special_reload_function:
205                            hits_to_proc_x = special_reload_function() or 0 # Refund function to reset after reload
206                            if special_reload_function_y:
207                                hits_to_proc_y = special_reload_function_y() or 0
208                        else:
209                            self.sim_state.time += self.reload_time
210                            if print_update:
211                                print(f"      - Reloading {self.reload_time} | Damage: {dmg}")
212                    else:
213                        self.sim_state.time += self.time_between_shots
214                        if print_update:
215                            print(f"      - Between shots {self.time_between_shots} | Damage: {dmg}")
216                if self.sim_state.shots_fired == self.reserves:
217                    break
218                self.sim_state.mag_size = self.mag_size_subsequent
219                self.sim_state.mag += 1
220    def processBaitDamageLoop(self, bait_tuple, damage_per_shot_function, dont_reproc = False, custom_proc_bait = None):
221        """
222        Simulates a damage loop using the BAIT mechanic.
223
224        Args:
225            bait_tuple (list): A 3-tuple of (swap_time, damage) representing the swap sequence (P→S, S→H, H→P).
226            damage_per_shot_function (Callable): Damage function with `is_proc_shot` flag.
227            dont_reproc (bool): If True, disables BAIT reproc (useful for envious high mag gls).
228            custom_proc_bait (Callable): Optional override function for handling BAIT procs (such as shooting a mag of a another weapon when swapping).
229        """
230        mag_size = self.mag_size_initial
231        
232        def procBait():
233            if print_update:
234                print(f"---------------------") 
235                print(f"      - Proccing bait")              
236            for index in range(2):
237                self.sim_state.damage_done += bait_tuple[index][1]
238                self.sim_state.damage_times.append(self.update())
239                if print_update:
240                  print(f"      - Adding Time {bait_tuple[index][0]}")
241                self.sim_state.time += bait_tuple[index][0]    
242            if print_update:
243                print(f"      - Proccing bait at {self.sim_state.time}")     
244                print(f"---------------------")  
245            return self.sim_state.time
246        procBait = custom_proc_bait or procBait
247        self.sim_state.last_bait_time = procBait() 
248        self.sim_state.procs = 1
249        self.sim_state.mag = 1
250        is_proc_shot = True
251        while self.sim_state.shots_fired < self.reserves and self.sim_state.time < 100:
252            for shots_fired_this_mag in range(mag_size):
253                self.sim_state.shots_fired_this_mag = shots_fired_this_mag
254                if dont_reproc and self.sim_state.time > self.sim_state.last_bait_time + self.bait_duration:
255                    is_proc_shot = True
256                damage_per_shot = damage_per_shot_function(is_proc_shot)
257                self.sim_state.damage_done += damage_per_shot
258                self.sim_state.shots_fired += 1
259                self.sim_state.damage_times.append(self.update())                    
260                is_proc_shot = False
261                if(self.sim_state.shots_fired == self.reserves):
262                    if print_update:
263                      print(f"      - Between shots {self.time_between_shots} | Damage: {damage_per_shot}")   
264                    break    
265                #break out early situations
266                if not dont_reproc and (self.sim_state.time + self.time_between_shots > self.sim_state.last_bait_time + self.bait_duration) and (self.sim_state.shots_fired_this_mag != mag_size - 1 or mag_size == 1) and self.sim_state.shots_fired != self.reserves - 1:
267                    if self.sim_state.shots_fired_this_mag == mag_size-1:
268                        self.sim_state.time += self.reload_num_appear   
269                        if print_update:
270                            print(f"      - Reloading Ammo {self.reload_num_appear} | Damage: {damage_per_shot}")
271                    else: 
272                        if print_update:
273                            print(f"      - Reprocing | Damage: {damage_per_shot}")   
274                    self.sim_state.time+=bait_tuple[2][0]
275                    if print_update:
276                        print(f"      - Heavy To Primary {bait_tuple[2][0]}")   
277                    self.sim_state.last_bait_time = procBait()
278                    self.sim_state.procs += 1
279                    is_proc_shot = True
280                elif(self.sim_state.shots_fired_this_mag == mag_size-1):
281                    #if need to reload, but bait will run out after
282                    if self.sim_state.time + self.reload_time > self.sim_state.last_bait_time + self.bait_duration and self.sim_state.time < self.sim_state.last_bait_time + self.bait_duration and (mag_size != 1 or (mag_size == 1 and self.sim_state.shots_fired != self.reserves - 1)):
283                        new_time = self.sim_state.last_bait_time + self.bait_duration + (1/60)
284                        if self.reload_num_appear + self.sim_state.time > self.sim_state.last_bait_time + self.bait_duration:
285                            if print_update:
286                                print(f"      - Reloading Ammo {self.reload_num_appear} | Damage: {damage_per_shot}")
287                            new_time = self.sim_state.time + self.reload_num_appear
288                            self.sim_state.time = new_time
289                        else:
290                            old_time = self.sim_state.time
291                            self.sim_state.time = new_time
292                            if print_update:
293                                print(f"      - Stalling and Reload {new_time - old_time}")
294                        self.sim_state.time+=bait_tuple[2][0]
295                        if print_update:
296                            print(f"      - Heavy To Primary {bait_tuple[2][0]}")   
297                        self.sim_state.last_bait_time = procBait()
298                        self.sim_state.procs += 1
299                        is_proc_shot = True
300                    #if need to reload and bait wont run out after
301                    else:
302                        if print_update:
303                            print(f"      - Reloading {self.reload_time} | Damage: {damage_per_shot}")
304                        self.sim_state.time+=self.reload_time 
305                elif (self.sim_state.shots_fired_this_mag != mag_size-1) : #only if we didnt reload or proc bait
306                    if print_update:
307                        print(f"      - Between shots {self.time_between_shots} | Damage: {damage_per_shot}")          
308                    self.sim_state.time+=self.time_between_shots                
309            if(self.sim_state.shots_fired==self.reserves):
310                    break    
311            mag_size = self.mag_size_subsequent
312            self.sim_state.mag += 1

314    def processEnviousArsenalBaitDamageLoop(self, bait_tuple, damage_per_shot_function):
315        """
316        Simulates damage using Envious Arsenal logic with BAIT procs and weapon rotation.
317
318        Args:
319            bait_tuple (list): Contains (time, damage) values for the other two weapons used to trigger BAIT.
320            damage_per_shot_function (Callable): Function returning damage per shot.
321        """
322
323        # Initialize variables
324        mag_size = self.mag_size_initial  # Current magazine size
325        weapons_shot_for_bait = set()  # Tracks which weapons have been shot for BAIT
326        self.sim_state.last_bait_time = 0  # Tracks when BAIT was last triggered
327        total_bait_proc_time = bait_tuple[0][0] + bait_tuple[1][0] + bait_tuple[2][0]
328        
329        # Function to check if BAIT should be reprocced
330        def check_bait_proc():
331            """
332            Reprocs BAIT if the timer has expired and all three weapons have been used.
333            Resets the `weapons_shot_for_bait` set and updates `bait_time`.
334            """
335            if len(weapons_shot_for_bait) == 3:
336                self.sim_state.last_bait_time = self.sim_state.time
337                weapons_shot_for_bait.clear()
338                if print_update:
339                    print(f"      - Proccing bait at {self.sim_state.time}")
340                    print(f"---------------------")
341
342        # Function to simulate shooting other guns (required to reproc BAIT)
343        def shootOtherGuns():
344            """
345            Simulates shooting the two other weapons to reproc BAIT.
346            Updates damage, time, and `weapons_shot_for_bait` accordingly.
347            """
348            self.sim_state.procs += 1
349            if print_update:
350                print(f"---------------------")
351                print(f"      - Shooting other guns")
352
353            for weapon, damage_time in zip(["k", "e"], bait_tuple):  # Iterate over two weapons
354                self.sim_state.damage_done += damage_time[1]  # Add damage from the weapon
355                self.sim_state.damage_times.append(self.update())  # Log damage event
356                if print_update:
357                    print(f"      - Adding Time {damage_time[0]}")
358                if self.sim_state.time > self.sim_state.last_bait_time + self.bait_duration or self.sim_state.last_bait_time == 0:  # Check if BAIT can be reprocced
359                    weapons_shot_for_bait.add(weapon)
360                    check_bait_proc()
361                self.sim_state.time += damage_time[0]  # Increment time for shooting the weapon
362            if print_update:
363                print(f"      - Swapping to bait weapon at {self.sim_state.time}")
364                print(f"---------------------")
365            return self.sim_state.time
366
367        # Initial BAIT proc
368        shootOtherGuns()
369        weapons_shot_for_bait.add("h")  # Track this weapon usage
370        check_bait_proc()
371        self.sim_state.procs = 1  # Tracks the number of times BAIT has been reprocced
372        self.sim_state.mag = 1  # Tracks the current magazine
373        is_proc_shot = True  # Indicates if the shot is a proc shot
374        # Main loop to simulate shooting and reloading
375        while self.sim_state.shots_fired < self.reserves and self.sim_state.time < 100:  # Continue until reserves are depleted or time limit is reached
376            for shots_fired_this_mag in range(mag_size):  # Loop through all shots in the magazine
377                self.sim_state.shots_fired_this_mag = shots_fired_this_mag
378                # Check if BAIT timer has expired before this shot
379                if self.sim_state.time > self.sim_state.last_bait_time + self.bait_duration:
380                    if print_update:
381                        print(f"      - Bait Expired at {self.sim_state.last_bait_time + self.bait_duration}")
382                    remaining_shots = self.reserves - self.sim_state.shots_fired
383                    # Calculate the number of magazines needed
384                    mags_if_shoot = math.ceil((remaining_shots - 1) / mag_size)
385                    mags_if_reproc = math.ceil(remaining_shots / mag_size)
386                    mags_if_shoot = math.ceil((remaining_shots - 1) / mag_size)
387                    # Decide whether to shoot the shot or reproc
388                    if mags_if_shoot < mags_if_reproc or len(weapons_shot_for_bait) == 2:
389                        if print_update:
390                            print(f"      - Shooting to reduce mags at {self.sim_state.time}")
391                        weapons_shot_for_bait.add("h")  # Mark bait weapon used
392                        check_bait_proc()
393                        is_proc_shot = True  # Mark the next shot as a proc shot
394                    else:
395                        if print_update:
396                            print(f"      - Reproccing BAIT at {self.sim_state.time}")
397                        break
398
399                
400                # Calculate damage for this shot
401                damage_per_shot = damage_per_shot_function(is_proc_shot)
402                self.sim_state.damage_done += damage_per_shot
403                self.sim_state.shots_fired += 1
404                self.sim_state.damage_times.append(self.update())  # Log damage event
405                is_proc_shot = False  # Reset proc shot flag
406                if self.sim_state.shots_fired == self.reserves:
407                    break
408                # Update time for the shot or reload
409                if self.sim_state.shots_fired_this_mag == mag_size - 1:  # Last shot in the magazine
410                    if mag_size == 1:
411                        if print_update:
412                            print(f"      - Performing an Envious Reload Mag is Empty at {self.sim_state.time}")
413                    elif self.sim_state.time + bait_tuple[2][0] < self.sim_state.last_bait_time + self.bait_duration and self.sim_state.time + total_bait_proc_time > self.sim_state.last_bait_time + self.bait_duration:
414                        if print_update:
415                            print(f"      - Performing an Envious Reload Mag WITH STALL is Empty at {self.sim_state.time}")
416                        self.sim_state.time = self.sim_state.last_bait_time + self.bait_duration + (1/60)
417                    else:
418                        if print_update:
419                            print(f"      - Performing an Envious Reload Mag is Empty at {self.sim_state.time}")
420                else:
421                    self.sim_state.time += self.time_between_shots  # Add time between shots
422                    if print_update:
423                        print(f"      - Between shots {self.time_between_shots} | Damage: {damage_per_shot}")
424
425            # Exit if all reserves have been used
426            if self.sim_state.shots_fired == self.reserves:
427                break
428
429            # Update magazine size and simulate shooting other guns
430            mag_size = self.mag_size_subsequent
431            self.sim_state.mag  += 1
432            self.sim_state.time += bait_tuple[2][0]
433            shootOtherGuns()

435    def processRefundLoop(self, damage_per_shot_function):
436        """
437        Simulates a refund-based damage loop where shots are regained periodically.
438        Depending on the self.refund_shots value the number of shots to refund and shots refunded will change
439        Calls the SimpledamageLoop function to handle the actual damage calculation and shot processing.
440
441        Args:
442            damage_per_shot_function (Callable): Function to compute damage per shot.
443        """
444        mapping = {
445                    4: 2,
446                    3: 1,
447                    10: 7
448        }
449        def after_x():
450            self.sim_state.mag_size += mapping[self.refund_shots]
451            if print_update:
452                print("      - Refunding {mapping[self.refund_shots]} shots")
453            self.reserves += mapping[self.refund_shots]
454        def reload_func():
