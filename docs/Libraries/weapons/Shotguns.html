103    def __init__(self, damage_multiplier=1.2):
104        super().__init__(
105            name="Fortismo (" + ("FF" if damage_multiplier == 1.2 else "Vorpal" + "FTTC)"),

123        return self.base_damage * buff_perc

178class FourthHorseMan(Shotgun):
179    def __init__(self, is_hs=False, is_rain_of=False, is_dodge=False):
180        damage_type = "horseman_hs" if is_hs else "horseman_bs"
181        super().__init__(
182            name="Fourth Horseman" + ("HS" if is_hs else "BS"),
183            reserves=18,
184            time_between_shots=11.5/60,
185            reload_time=162/60,  # Default Lunas reload time
186            mag_size_initial=5,
187            mag_size_subsequent=5,
188            damage_type=damage_type,
189            damage_loop_type="simple"
190        )
191        self.rainOF_reload_time = 62/60
192        self.dodge_reload_time = 92/60
193        self.reload_time = self.rainOF_reload_time if is_rain_of else self.dodge_reload_time if is_dodge else self.reload_time
194        
195    def damage_per_shot_function(self, buff_perc):
196        if (self.sim_state.shots_fired_this_mag == 1):
197            return self.base_damage * 1.18 * buff_perc
198        elif (self.sim_state.shots_fired_this_mag == 2):
199            return self.base_damage * 1.39 * buff_perc
200        elif (self.sim_state.shots_fired_this_mag == 3):
201            return self.base_damage * 1.59 * buff_perc
202        elif (self.sim_state.shots_fired_this_mag == 4):
203            return self.base_damage * 1.81 * buff_perc
204        return self.base_damage * buff_perc

206class LordOfWolves(Shotgun):
207    def __init__(self, has_perk=True):
208        burst_cooldown = 8/60 if has_perk else 12/60
209        damage_type = "lordow_perk" if has_perk else "lordow"
210        super().__init__(
211            name="Lord of Wolves (Release the Wolves)" if has_perk else "Lord of Wolves",
212            reserves=30,
213            time_between_shots=4/60,
214            reload_time=burst_cooldown,  # Default reload time
215            mag_size_initial=5, #Burst size
216            mag_size_subsequent=5,
217            damage_type=damage_type,
218            damage_loop_type="simple"
219        )
220        self.real_reserves = 270
221        self.real_reload_time = 64/60
222        self.real_reload_time_perk = 58/60 
223        self.has_perk = has_perk
224    def calculate(self, buff_perc = 1.25, name="Lord Of Wolves", prev_result=None):
225        self._prepare_calculation(prev_result)
226
227        reload_time = self.real_reload_time_perk if self.has_perk else self.real_reload_time
228        def damage_per_shot_function():
229            return self.base_damage * buff_perc
230        while self.real_reserves > 0:
231            self.processSimpleDamageLoop(damage_per_shot_function)
232            self.sim_state.time += reload_time
233            self.real_reserves -= self.reserves # Reserves is mag size
234            self.sim_state.soft_reset()
235        return self.fill_gaps(self.sim_state.damage_times, name, self.category)

Attributes: name (str): Name of the weapon. reserves (int): Total shots available in reserves. charge_time (float): Time before the weapon starts firing. time_between_shots (float): Delay between consecutive shots. reload_time (float): Time taken to reload the weapon. mag_size_initial (int): Initial magazine size. mag_size_subsequent (int): Magazine size after the first reload. category (str): Required label for the weapon type (default is "h" for Heavy). damage_loop_type (str): Specifies which damage loop logic to use ('simple', 'bait', 'envious', 'refund'). refund_shots (int): Number of shots to refund in a refund loop. refund_progress_per_shot (int): Progress toward refunds per shot fired. bait_duration (float): Duration for which the BAIT buff is active. buffs (dict): Mapping of all supported damage buffs. damage_values (dict): Mapping of damage types to base damage values. sim_state (SimulationState): Tracks the current state of the simulation.
