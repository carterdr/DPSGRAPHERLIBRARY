233    """Anarchy Grenade Launcher dealing tick-based damage over time."""
234
235    def __init__(self):
236        super().__init__(name="Anarchy",
237                         reserves=25,
238                         reload_time=129 / 60,

240                         mag_size_initial=6,

242                         damage_type="anarchy",
243                         damage_loop_type="dot")
244
245    def calculate(self, buff_perc=1.25, prev_result=None, custom_name=None):
246        """Calculates Anarchys tick-based damage over time."""
247        name = custom_name or self.name
248        self._prepare_calculation(prev_result)
249        ticks = 0
250
251        while self.sim_state.time <= self.MAX_SIM_TIME:
252            if ticks == 0:
253                self.sim_state.damage_done += self.damage_values["anarchy_initial"] * buff_perc
254            self.sim_state.damage_done += self.damage_values["anarchy_tick"] * buff_perc
255            ticks += 1
256
257            if ticks % 19 == 0:
258                self.sim_state.damage_done += self.damage_values["anarchy_end"] * buff_perc
259                ticks = 0  # Reset tick cycle
260            self.sim_state.shots_fired = ticks
261            self.sim_state.damage_times.append(self.update())
262            self.sim_state.time += self.time_between_shots
263
264        return self.fill_gaps(self.sim_state.damage_times, name, self.category)

376        # Setup Choir of One
